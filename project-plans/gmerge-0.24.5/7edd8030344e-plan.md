      extension: null,
      extensionManager: null,

    const parser = yargs([]).command(settingsCommand);
    
    await parser.parseAsync('settings set my-extension apiKey myvalue');

  it('should show error and exit when extension config not found', async () => {
    mockGetExtensionAndManager.mockResolvedValue({
      extension: { name: 'my-extension' },

      extensionConfig: null,

    const parser = yargs([]).command(settingsCommand);
    
    await parser.parseAsync('settings set my-extension apiKey myvalue');

    expect(mockExitCli).toHaveBeenCalled();

  it('should update setting successfully', async () => {
    const mockExtension = {
      name: 'my-extension',

      name: 'my-extension',

        {

          description: 'API key',
          sensitive: true,

      extension: mockExtension,

      extensionConfig: mockConfig,
