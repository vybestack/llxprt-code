## What Upstream Does

Enables SessionStart hooks to inject additional context into the session via `hookSpecificOutput.additionalContext`. In non-interactive mode, context is prepended to the input prompt. In interactive mode, context is added to chat history as a user message. The hook also returns a `systemMessage` that's displayed to the user. The trigger function `triggerSessionStartHook()` now returns `DefaultHookOutput | undefined` instead of `void`. CLI entry points (gemini.tsx, AppContainer.tsx, clearCommand.ts) consume the result and inject context accordingly.

**Change implementation:**
```typescript
export async function triggerSessionStartHook(
  config: Config,
  source: SessionStartSource,
): Promise<DefaultHookOutput | undefined> {
  const hookSystem = config.getHookSystem();
  if (!hookSystem?.isInitialized()) {
    return undefined;
  }

  const eventHandler = hookSystem.getEventHandler();
  const hookInput: SessionStartInput = {
    session_id: config.getSessionId(),
    transcript_path: config.getTranscriptPath(),
    cwd: config.getProjectRoot(),
    hook_event_name: HookEventName.SessionStart,
    timestamp: new Date().toISOString(),
    source,
  };

  const response = await eventHandler.fireEvent(HookEventName.SessionStart, hookInput);
  
  if (response?.output) {
    return createHookOutput(HookEventName.SessionStart, response.output);
  }
  
  return undefined;
}
```

if (hasPipedInput) {
  const stdinData = await readStdinOnce();
  if (stdinData) {
    input = `${stdinData}\n\n${input}`;
  }
}

// Fire SessionStart hook and inject context
const sessionStartOutput = await triggerSessionStartHook(config, SessionStartSource.Startup);
if (sessionStartOutput) {
  // Display system message
  if (sessionStartOutput.systemMessage) {
    writeToStderr(`${sessionStartOutput.systemMessage}\n`);
  }
  
  // Prepend additional context
  const additionalContext = sessionStartOutput.getAdditionalContext();
  if (additionalContext) {
    input = `${additionalContext}\n\n${input}`;
  }
}

**OLD:**

**NEW:**
```typescript
useEffect(() => {
  const initializeSession = async () => {
    // ... existing initialization ...
    
    const sessionStartOutput = await triggerSessionStartHook(config, SessionStartSource.Startup);
    
    // Display system message
    if (sessionStartOutput?.systemMessage) {
      addItem({
        type: 'info',
        message: sessionStartOutput.systemMessage,
      });
    }
    
    // Add additional context to history
    const additionalContext = sessionStartOutput?.getAdditionalContext();
    if (additionalContext) {
      const geminiClient = config.getGeminiClient();
      if (geminiClient) {
        await geminiClient.addHistory({
          role: 'user',
          parts: [{ text: additionalContext }],
        });
      }
    }
    
    // ... rest of initialization ...
  };
  initializeSession();
}, []);
```
