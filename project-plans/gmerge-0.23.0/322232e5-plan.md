   ```typescript
   /**
    * Detects terminal background color using OSC 11 escape sequence
    * Returns hex color string like '#1E1E2E' or undefined if detection fails
    */
   export function detectTerminalBackgroundColor(): Promise<string | undefined> {
     return new Promise((resolve) => {
       const stdin = process.stdin;
       if (!stdin.isTTY) {
         resolve(undefined);
         return;
       }

       let timeout: NodeJS.Timeout;
       let response = '';

       const cleanup = () => {
         stdin.setRawMode(false);
         stdin.removeListener('data', dataHandler);
         clearTimeout(timeout);
       };

       const dataHandler = (data: Buffer) => {
         response += data.toString();
         // OSC 11 response format: ESC ] 11 ; rgb:RRRR/GGGG/BBBB ESC \
         const match = response.match(/\x1b\]11;rgb:([0-9a-f]{4})\/([0-9a-f]{4})\/([0-9a-f]{4})\x1b\\/i);
         if (match) {
           const r = parseInt(match[1].substring(0, 2), 16);
           const g = parseInt(match[2].substring(0, 2), 16);
           const b = parseInt(match[3].substring(0, 2), 16);
           cleanup();
           resolve(`#${r.toString(16).padStart(2, '0')}${g.toString(16).padStart(2, '0')}${b.toString(16).padStart(2, '0')}`.toUpperCase());
         }
       };

       stdin.on('data', dataHandler);
       stdin.setRawMode(true);

       // Query background color
       process.stdout.write('\x1b]11;?\x1b\\');

       // Timeout after 100ms
       timeout = setTimeout(() => {
         cleanup();
         resolve(undefined);
       }, 100);
     });
   }

   /**
    * Resolves color value (handles hex colors, returns as-is)
    */
   export function resolveColor(color: string | undefined): string | undefined {
     return color;
   }

   /**
    * Calculates relative luminance and returns 'light' or 'dark'
    */
   export function getThemeTypeFromBackgroundColor(
     bgColor: string | undefined,
   ): 'light' | 'dark' | undefined {
     if (!bgColor) return undefined;

     const hex = bgColor.replace('#', '');
     if (hex.length !== 6) return undefined;

     const r = parseInt(hex.substring(0, 2), 16) / 255;
     const g = parseInt(hex.substring(2, 4), 16) / 255;
     const b = parseInt(hex.substring(4, 6), 16) / 255;

     // sRGB to linear RGB
     const toLinear = (c: number) => c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);
     const lr = toLinear(r);
     const lg = toLinear(g);
     const lb = toLinear(b);

     // Calculate relative luminance
     const luminance = 0.2126 * lr + 0.7152 * lg + 0.0722 * lb;

     return luminance > 0.5 ? 'light' : 'dark';
   }
   ```

2. **Add pickDefaultThemeName** to packages/cli/src/ui/themes/theme.ts:
   ```typescript
   import { getThemeTypeFromBackgroundColor } from './color-utils.js';

   /**
    * Picks a default theme name based on terminal background color
    */
   export function pickDefaultThemeName(
     terminalBackgroundColor: string | undefined,
     availableThemes: { name: string; type: string }[],
     fallbackTheme: string,
     fallbackLightTheme?: string,
   ): string {
     const terminalThemeType = getThemeTypeFromBackgroundColor(terminalBackgroundColor);
     
     if (!terminalThemeType) {
       return fallbackTheme;
     }

     // Find first theme matching terminal type
     const matchingTheme = availableThemes.find(t => t.type === terminalThemeType);
     if (matchingTheme) {
       return matchingTheme.name;
     }

     // Fallback to light theme if terminal is light
     if (terminalThemeType === 'light' && fallbackLightTheme) {
       return fallbackLightTheme;
     }

     return fallbackTheme;
   }
   ```
